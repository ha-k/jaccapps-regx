package hlt.regx;

import java.io.File;

public class Main
{
  public static void main (String[] args)
    {
      try
        {
          Tokenizer t;

          if(args.length>0)
	    {
	      t = new Tokenizer(new File(args[0]));
	    }
          else
            t = new Tokenizer();

          try
            {
              new Parser(t).parse();
            }
          catch (Exception e)
            {
              System.out.println("*** Parsing error: "+e);
	      e.printStackTrace();
            }
        }
      catch (Exception e)
        {
          System.out.println("*** Tokenizing error: "+e);
        }
    }
}
